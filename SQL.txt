SELECT c.login, COUNT(o.id) AS "deliveryCount" 
FROM "Couriers" AS c 
JOIN "Orders" AS o ON c.id = o."courierId" 
WHERE o."inDelivery" = true 
GROUP BY c.login;

2.
SELECT track, 
  case 
    WHEN finished = true then 2 
    WHEN cancelled = true then -1 
    WHEN "inDelivery" = true then 1 
    ELSE 0 
  END as "Order status" 
FROM "Orders";
